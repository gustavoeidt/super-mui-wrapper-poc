import { Meta } from "@storybook/addon-docs";
import { colors } from "../../themes/light/colors";

export const ColorBlock = ({ backgroundColor, color }) => {
  const copyToClipboard = () => {
    navigator.clipboard.writeText(backgroundColor);
  };
  const blockStyle = {
    backgroundColor: backgroundColor,
    padding: "1rem",
    color: color,
    border: `0.01rem solid ${color}`,
  };
  return (
    <div style={blockStyle} onClick={copyToClipboard}>
      {backgroundColor}
    </div>
  );
};
